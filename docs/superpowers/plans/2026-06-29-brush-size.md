# Brush Size Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give the pencil and eraser a 1–4 px square brush size, cycled by tapping the already-selected tool, with the toolbar glyph showing the current nib.

**Architecture:** A new pure `Brush` in the Drawing layer returns the cells a square nib covers. `EditorState` holds a per-tool size and a `stampBrush` that paints the nib (mirror per-cell, pixel-perfect only at size 1) inside one undo snapshot. `LineTool` gains an optional `brushSize`. The freehand path (`applyDrawPoint`) and the hold-to-straighten line both route through these. The `ToolBar` glyph becomes the nib (pencil filled, eraser hollow).

**Tech Stack:** Swift, SwiftUI, XCTest. Xcode project generated by xcodegen (`DrawBit.xcodeproj` is gitignored — never commit it).

## Global Constraints

- **Pure-logic rule:** `DrawBit/Drawing/Brush.swift` imports only `Foundation`. No SwiftUI/SwiftData/UIKit.
- **Straight alpha, 0/255 only:** stamps write opaque `color` (pencil) or `RGBA.transparent` (eraser). Never partial alpha.
- **One undo step per stroke:** every stamped + mirrored cell writes inside a single `mutateActiveLayerPixels` closure.
- **Session-scoped sizes:** `pencilSize`/`eraserSize` live on `EditorState`, default 1, never written to `Piece` or `AppSettings`.
- **Range via constant:** `brushSizeRange = 1...4`; the cycle wrap reads `brushSizeRange.upperBound`, never a literal `4`.
- **Regenerate the project after adding/renaming a `.swift` file:** `rm -rf DrawBit.xcodeproj && xcodegen generate`.

import Foundation

/// Square brush-nib geometry. Pure: the canvas cells an `n × n` nib covers when its center is
/// dropped on `point`, row-major. Odd sizes center on the cursor; even sizes put the cursor at the
/// upper-left of the central 2×2 (the nib extends one extra pixel down and right). Cells may fall
/// off-canvas — callers paint through `PixelGrid.setPixel`, which silently drops out-of-bounds
/// writes, so there is no clamping here.
enum Brush {
    static func squareCells(centeredAt point: (Int, Int), size: Int) -> [(Int, Int)] {
        let n = max(1, size)
        let originX = point.0 - (n - 1) / 2
        let originY = point.1 - (n - 1) / 2
        var cells: [(Int, Int)] = []
        cells.reserveCapacity(n * n)
        for j in 0..<n {
            for i in 0..<n {
                cells.append((originX + i, originY + j))
            }
        }
        return cells
    }
}
```

    func stampBrush(_ tool: Tool, at point: (Int, Int)) {
        let n = brushSize(for: tool)
        let cells = Brush.squareCells(centeredAt: point, size: n)
        let w = size.width - 1
        let paintColor = color
        mutateActiveLayerPixels { data in
            var grid = PixelGrid(data: data, size: size)
            if n == 1 {
                revertPixelPerfectElbow(grid: &grid, beforeApplyingAt: point)
            }
            func write(_ x: Int, _ y: Int) {
                switch tool {
                case .pencil: Pencil.paint(on: &grid, at: (x, y), color: paintColor)
                case .eraser: Eraser.erase(on: &grid, at: (x, y))
                default: break
                }
            }
            for (cx, cy) in cells {
                write(cx, cy)
                if isMirrorEnabled {
                    let mx = w - cx
                    if mx != cx { write(mx, cy) }
                }
            }
            data = grid.data
        }
    }
```

    static func apply(to base: Data, size: CanvasSize, from: (Int, Int), to: (Int, Int),
                      color: RGBA, mirror: Bool, brushSize: Int = 1) -> Data {
        var grid = PixelGrid(data: base, size: size)
        stampLine(on: &grid, from: from, to: to, color: color, brushSize: brushSize)
        if mirror {
            let w = size.width - 1
            stampLine(on: &grid, from: (w - from.0, from.1), to: (w - to.0, to.1),
                      color: color, brushSize: brushSize)
        }
        return grid.data
    }

    private static func stampLine(on grid: inout PixelGrid, from: (Int, Int), to: (Int, Int),
                                  color: RGBA, brushSize: Int) {
        if brushSize <= 1 {
            Pencil.paintLine(on: &grid, from: from, to: to, color: color)
            return
        }
        for p in bresenhamLine(from: from, to: to) {
            for (cx, cy) in Brush.squareCells(centeredAt: p, size: brushSize) {
                grid.setPixel(x: cx, y: cy, color: color)
            }
        }
    }

    }

    /// Nib size 1...4 → glyph side in points.
    private func nibGlyphSide(for size: Int) -> CGFloat {
        let sides: [CGFloat] = [8, 12, 16, 20]
        return sides[min(max(size, 1), 4) - 1]
    }
